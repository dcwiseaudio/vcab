async function startWithQN(){ const qnRaw = document.getElementById('qnInput').value.trim(); const err = document.getElementById('introError'); err.textContent = ''; err.classList.add('hidden'); if (!qnRaw) { err.textContent = "QN을 입력해 주세요."; err.classList.remove('hidden'); return; } const qn = qnRaw.toLowerCase(); const ok = await loadFromSheet(qn); if (!ok) return; // ✅ QN을 URL 파라미터로 넘기고 "페이지 이동" location.href = `menu.html?qn=${encodeURIComponent(qn)}`; }
